\subsubsection{Data Wrangling}\label{data-wrangling}}

    \hypertarget{data-description}{%
\paragraph{Data Description}\label{data-description}}

    The Cornell Movie Dialog corpus contains a metadata-rich collection of
fictional conversations extracted from raw movie scripts:

\begin{itemize}
\tightlist
\item
  220,579 conversational exchanges between 10,292 pairs of movie
  characters
\item
  involves 9,035 characters from 617 movies
\item
  in total 304,713 utterances
\item
  movie metadata included:

  \begin{itemize}
  \tightlist
  \item
    genres
  \item
    release year
  \item
    IMDB rating
  \item
    number of IMDB votes
  \item
    IMDB rating
  \end{itemize}
\item
  character metadata included:

  \begin{itemize}
  \tightlist
  \item
    gender (for 3,774 characters)
  \item
    position on movie credits (3,321 characters)
  \end{itemize}
\end{itemize}

\begin{enumerate}
\def\labelenumi{\Alph{enumi})}
\setcounter{enumi}{1}
\tightlist
\item
  Files description:
\end{enumerate}

In all files the field separator is " +++\$+++ "

\begin{itemize}
\tightlist
\item
  movie\_titles\_metadata.txt

  \begin{itemize}
  \tightlist
  \item
    contains information about each movie title
  \item
    fields:

    \begin{itemize}
    \tightlist
    \item
      movieID,
    \item
      movie title,
    \item
      movie year,
    \item
      IMDB rating,
    \item
      no. IMDB votes,
    \item
      genres in the format {[}`genre1',`genre2',�,`genreN'{]}
    \end{itemize}
  \end{itemize}
\item
  movie\_characters\_metadata.txt

  \begin{itemize}
  \tightlist
  \item
    contains information about each movie character
  \item
    fields:

    \begin{itemize}
    \tightlist
    \item
      characterID
    \item
      character name
    \item
      movieID
    \item
      movie title
    \item
      gender (``?'' for unlabeled cases)
    \item
      position in credits (``?'' for unlabeled cases)
    \end{itemize}
  \end{itemize}
\item
  movie\_lines.txt

  \begin{itemize}
  \tightlist
  \item
    contains the actual text of each utterance
  \item
    fields:

    \begin{itemize}
    \tightlist
    \item
      lineID
    \item
      characterID (who uttered this phrase)
    \item
      movieID
    \item
      character name
    \item
      text of the utterance
    \end{itemize}
  \end{itemize}
\item
  movie\_conversations.txt

  \begin{itemize}
  \tightlist
  \item
    the structure of the conversations
  \item
    fields

    \begin{itemize}
    \tightlist
    \item
      characterID of the first character involved in the conversation
    \item
      characterID of the second character involved in the conversation
    \item
      movieID of the movie in which the conversation occurred
    \item
      list of the utterances that make the conversation, in
      chronological order: {[}`lineID1',`lineID2',�,`lineIDN'{]} has to
      be matched with movie\_lines.txt to reconstruct the actual content
    \end{itemize}
  \end{itemize}
\item
  raw\_script\_urls.txt

  \begin{itemize}
  \tightlist
  \item
    the urls from which the raw sources were retrieved
  \end{itemize}
\end{itemize}
